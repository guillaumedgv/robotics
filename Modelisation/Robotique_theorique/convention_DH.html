import numpy as np
import matplotlib.pyplot as plt


#Matrice MDH
def DH(alpha1, d1, theta1, r1):
    T = np.array([
        [np.cos(theta1),            -np.sin(theta1),            0,  d1],
        [np.cos(alpha1)*np.sin(theta1), np.cos(alpha1)*np.cos(theta1), -np.sin(alpha1), -r1*np.sin(alpha1)],
        [np.sin(alpha1)*np.sin(theta1), np.sin(alpha1)*np.cos(theta1),  np.cos(alpha1),  r1*np.cos(alpha1)],
        [0, 0, 0, 1]
    ])
    return T


# Vecteur des angles articulaires
q = np.array([0, 0, 0, 0, 0, 0, 0])

# Matrice des param√®tres DH
DH_parameters = np.array([
    [0,        0,        q[0], 0.317],
    [-np.pi/2, 0,        q[1], 0],
    [ np.pi/2, 0,        q[2], 0.45],
    [-np.pi/2, 0,        q[3], 0],
    [ np.pi/2, 0,        q[4], 0.48],
    [-np.pi/2, 0,        q[5], 0],
    [ np.pi/2, 0,        q[6], 0.07]
])

# -----------------------------
# MGD
# -----------------------------
T = np.eye(4)

    Ti = DH(alpha, d, theta, r)
    T = T @ Ti
    points.append(T[:3, 3])   # sauvegarde pour le plot
